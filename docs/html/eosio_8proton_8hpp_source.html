194 static std::map<std::string,uint8_t> get_priv( name contract_account, name acc ){
195 std::map<std::string,uint8_t> res;
196
197 //exception for eosio account
198 if ( acc == "eosio"_n ) {
199 res["createacc"] = 1; res["vote"] = 1; res["regprod"] = 1; res["regproxy"] = 1; res["setcontract"] = 1; res["namebids"] = 1; res["rex"] = 1; res["delegate"] = 1; res["undelegate"] = 1; res["sellram"] = 1; res["buyram"] = 1;
200 return res;
201 }
202
203 res["createacc"] = 0; res["vote"] = 0; res["regprod"] = 0; res["regproxy"] = 0; res["setcontract"] = 0; res["namebids"] = 0; res["rex"] = 0; res["delegate"] = 0; res["undelegate"] = 0; res["sellram"] = 0; res["buyram"] = 0;
204
205 permissions perm( contract_account, contract_account.value );
206 auto existing = perm.find( acc.value );
207 if ( existing != perm.end() ) {
208 res["createacc"] = existing->createacc;
209 res["vote"] = existing->vote;
210 res["regprod"] = existing->regprod;
211 res["regproxy"] = existing->regproxy;
212 res["setcontract"] = existing->setcontract;
213 res["namebids"] = existing->namebids;
214 res["rex"] = existing->rex;
215
216 res["delegate"] = existing->delegate;
217 res["undelegate"] = existing->undelegate;
218 res["sellram"] = existing->sellram;
219 res["buyram"] = existing->buyram;
220
221 }
222 return res;
223 }
