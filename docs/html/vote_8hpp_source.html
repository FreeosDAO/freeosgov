75std::vector<int> parse_vote_ranges(string voteranges) {
76
77 // the voteranges string looks like this: q1:0-100,q2:6-30,q5:0-50
78 std::vector<int> limits;
79
80 std::vector<std::string> tokenlist = split(voteranges, ",");
81
82 std::vector q1_param = split(tokenlist[0], ":");
83 std::vector q2_param = split(tokenlist[1], ":");
84 std::vector q5_param = split(tokenlist[2], ":");
85
86 std::vector q1_minmax = split(q1_param[1], "-");
87 std::vector q2_minmax = split(q2_param[1], "-");
88 std::vector q5_minmax = split(q5_param[1], "-");
89
90 limits.push_back (stoi(q1_minmax[0]));
91 limits.push_back (stoi(q1_minmax[1]));
92 limits.push_back (stoi(q2_minmax[0]));
93 limits.push_back (stoi(q2_minmax[1]));
94 limits.push_back (stoi(q5_minmax[0]));
95 limits.push_back (stoi(q5_minmax[1]));
96
97 return limits;
98}

125
126 require_auth(user);
127
128 tick();
129
130 // check that system is operational (masterswitch parameter set to "1")
131 check(check_master_switch(), MSG_FREEOS_SYSTEM_NOT_AVAILABLE);
132
133 // is the user registered?
134 check(is_registered(user), "voting is not open to unregistered users");

138
139 // is the system operational?

141 check(this_iteration != 0, "The freeos system is not available at this time");
142
143 // is the user alive?
144 check(is_user_alive(user), "user has exceeded the maximum number of iterations");
145
146 // are we in the vote period?
147 check(is_action_period("vote"), "It is outside of the vote period");
148
149 // has the user already completed the vote?
150 svr_index svrs_table(get_self(), user.value);
151 auto svr_iterator = svrs_table.begin();
152
153 // if there is no svr record for the user then create it - we will update it at the end of the action
154 if (svr_iterator == svrs_table.end()) {
155 // emplace
156 svrs_table.emplace(get_self(), [&](auto &svr) { ; });
157 svr_iterator = svrs_table.begin();
158 } else {
159 check(svr_iterator->vote0 != this_iteration &&
160 svr_iterator->vote1 != this_iteration &&
161 svr_iterator->vote2 != this_iteration &&
162 svr_iterator->vote3 != this_iteration &&
163 svr_iterator->vote4 != this_iteration,
164 "user has already voted");
165 }
166
167 // parameter checking
168 // n.b. split function is defined in survey.hpp
169
170 // get and parse the vote slider ranges
171 string voteranges = get_parameter(name("voteranges"));
172 std::vector<int> vote_range_values = parse_vote_ranges(voteranges);
173
174 // get the current price of Freeos
175 exchange_index rates_table(get_self(), get_self().value);
176 auto rate_iterator = rates_table.begin();
177 check(rate_iterator != rates_table.end(), "current price of Freeos is undefined");
178 double current_price = rate_iterator->currentprice;
179 // double target_price = rate_iterator->targetprice;
180
181 // calculate the upper bound of locking threshold (q3)
182 double lock_factor = get_dparameter(name("lockfactor"));
183
184 double locking_threshold_upper_limit = 0.0;
185 if (current_price < HARD_EXCHANGE_RATE_FLOOR) {
186 locking_threshold_upper_limit = lock_factor * HARD_EXCHANGE_RATE_FLOOR;
187 } else {
188 locking_threshold_upper_limit = lock_factor * current_price;
189 }
190
191
192 check(q1response >= vote_range_values[0] && q1response <= vote_range_values[1], "Response 1 is out of range");
193 check(q2response >= vote_range_values[2] && q2response <= vote_range_values[3], "Response 2 (FREEOS mint fee) is out of range");
194 check(q2response_xpr >= vote_range_values[2] && q2response_xpr <= vote_range_values[3], "Response 2 (XPR mint fee) is out of range");
195 check(q2response_xusdc >= vote_range_values[2] && q2response_xusdc <= vote_range_values[3], "Response 2 (XUSDC mint fee) is out of range");
196 check(q3response >= HARD_EXCHANGE_RATE_FLOOR && q3response <= locking_threshold_upper_limit, "Response 3 is out of range");
197 check(q4response == "POOL" || q4response == "BURN", "Response 4 must be 'POOL' or 'BURN'");
198 check(q5response >= vote_range_values[4] && q5response <= vote_range_values[5], "Response 5 is out of range");
199 check(q6choice1 >= 1 && q6choice1 <= 6, "Response 6 choice 1 must be a number between 1 and 6");
200 check(q6choice2 >= 1 && q6choice2 <= 6, "Response 6 choice 2 must be a number between 1 and 6");
201 check(q6choice3 >= 1 && q6choice3 <= 6, "Response 6 choice 3 must be a number between 1 and 6");
202
203 // response 6 - the 3 choices must not contain duplicates
204 check((q6choice1 != q6choice2) && (q6choice2 != q6choice3) && (q6choice3 != q6choice1), "Response 6 has duplicate values");
205
206 // store the responses
207 vote_index vote_table(get_self(), get_self().value);
208 auto vote_iterator = vote_table.begin();
209 check(vote_iterator != vote_table.end(), "vote record is not defined");
210
211 // process the responses from the user
212 // for multiple choice options, increment to add the user's selection
213 // for running averages, compute new running average
214 vote_table.modify(vote_iterator, get_self(), [&](auto &vote) {
215
216 // question 1
217 vote.q1average = ((vote.q1average * vote.participants) + q1response) / (vote.participants + 1);
218
219 // question 2
220 vote.q2average = ((vote.q2average * vote.participants) + q2response) / (vote.participants + 1);
221 vote.q2average_xpr = ((vote.q2average_xpr * vote.participants) + q2response_xpr) / (vote.participants + 1);
222 vote.q2average_xusdc = ((vote.q2average_xusdc * vote.participants) + q2response_xusdc) / (vote.participants + 1);
223
224 // question 3
225 vote.q3average = ((vote.q3average * vote.participants) + q3response) / (vote.participants + 1);
226
227 // question 4
228 if (q4response == "POOL") {
229 vote.q4choice1++;
230 } else {
231 vote.q4choice2++;
232 }
233
234 // question 5
235 vote.q5average = ((vote.q5average * vote.participants) + q5response) / (vote.participants + 1);
236
237 // question 6 - need to iterate the list of choices
238 uint8_t q6choices[3] = { q6choice1, q6choice2, q6choice3 };
239 uint8_t points[3] = { 3,2,1 }; // points to distribute for 1st, 2nd, 3rd choices
240
241 for (size_t i = 0; i < sizeof(q6choices); i++) {
242 switch(q6choices[i]) {
243 case 1:
244 vote.q6choice1 += points[i];
245 break;
246 case 2:
247 vote.q6choice2 += points[i];
248 break;
249 case 3:
250 vote.q6choice3 += points[i];
251 break;
252 case 4:
253 vote.q6choice4 += points[i];
254 break;
255 case 5:
256 vote.q6choice5 += points[i];
257 break;
258 case 6:
259 vote.q6choice6 += points[i];
260 break;
261 }
262 }
263
264 // update the number of participants
265 vote.participants += 1;
266
267 }); // end of modify
268
269 // record that the user has responded to this iteration's vote
270 uint32_t survey_completed = 0;
271 svrs_table.modify(svr_iterator, get_self(), [&](auto &svr) {
272 switch (this_iteration % 5) {
273 case 0: svr.vote0 = this_iteration; survey_completed = svr.survey0; break;
274 case 1: svr.vote1 = this_iteration; survey_completed = svr.survey1; break;
275 case 2: svr.vote2 = this_iteration; survey_completed = svr.survey2; break;
276 case 3: svr.vote3 = this_iteration; survey_completed = svr.survey3; break;
277 case 4: svr.vote4 = this_iteration; survey_completed = svr.survey4; break;
278 }
279 }); // end of modify
280
281 // increment the number of participants in this iteration...
282 // ... unless they have completed survey, in which case they have already been counted
283 if (survey_completed != this_iteration) {
284 // increment the number of participants in this iteration
285 system_index system_table(get_self(), get_self().value);
286 auto system_iterator = system_table.begin();
287 check(system_iterator != system_table.end(), "system record is undefined");
288 system_table.modify(system_iterator, get_self(), [&](auto &s) {
289 s.participants += 1;
290 });
291 }
292
293 // update the votes counter in the user's participant record
294 participants_index participants_table(get_self(), user.value);
295 auto participant_iterator = participants_table.begin();
296 check(participant_iterator != participants_table.end(), "participant record not found");
297 participants_table.modify(participant_iterator, get_self(), [&](auto &p) {
298 p.votes += 1;
299 });
300
301}
