55void freeosgov::issue(const name &to, const asset &quantity, const string &memo) {
56 auto sym = quantity.symbol;
57 check(sym.is_valid(), "invalid symbol name");
58 check(memo.size() <= 256, "memo has more than 256 bytes");
59
60 stats statstable(get_self(), sym.code().raw());
61 auto existing = statstable.find(sym.code().raw());
62 check(existing != statstable.end(),
63 "token with symbol does not exist, create token before issue");
64 const auto &st = *existing;
65 check(to == st.issuer, "tokens can only be issued to issuer account");
66
67 check(quantity.is_valid(), "invalid quantity");
68 check(quantity.amount > 0, "must issue positive quantity");
69
70 check(quantity.symbol == st.supply.symbol, "symbol precision mismatch");
71 check(quantity.amount <= st.max_supply.amount - st.supply.amount,
72 "quantity exceeds available supply");
73
74 statstable.modify(st, same_payer, [&](auto &s) { s.supply += quantity; });
75
76 add_balance(st.issuer, quantity, st.issuer);
77}

147void freeosgov::unlock(const name &user) {
148 require_auth(user);
149
150 // user-activity-driven background process
151 tick();
152
153 // check that system is operational masterswitch parameter set to "1")
154 check(check_master_switch(), MSG_FREEOS_SYSTEM_NOT_AVAILABLE);
155
156 // get the current iteration
157 uint32_t this_iteration = current_iteration();
158 check(this_iteration > 0, "unlocking is not possible at this time, please try later");
159
160 // calculate the amount to be unvested - get the percentage for the iteration
161 system_index system_table(get_self(), get_self().value);
162 auto system_iterator = system_table.begin();
163 check(system_iterator != system_table.end(), "system record not found");
164 uint32_t unlock_percent = system_iterator->unlockpercent;
165
166 // check that the unvest percentage is within limits
167 check(unlock_percent > 0 && unlock_percent <= 100,
168 "locked POINTs cannot be unlocked in this claim period. Please try during next claim period");
169
170 // has the user unlocked in this iteration? - consult the unvest history table
171 unvest_index unvest_table(get_self(), user.value);
172 auto unvest_iterator = unvest_table.begin();
173 // if the unvest record exists for the current iteration then the user has already unvested,
174 // so is not eligible to unvest again
175 if (unvest_iterator != unvest_table.end()) {
176 check(unvest_iterator->iteration_number != this_iteration,
177 "user has already unlocked in this iteration");
178 }
179
180 // do the unlocking
181 // get the user's unvested POINT balance
182 asset locked_balance = asset(0, POINT_CURRENCY_SYMBOL);
183 lockaccounts locked_accounts_table(get_self(), user.value);
184 auto locked_account_iterator = locked_accounts_table.begin();
185
186 if (locked_account_iterator != locked_accounts_table.end()) {
187 locked_balance = locked_account_iterator->balance;
188 }
189
190 // if user's locked balance is 0 then nothing to do, so return
191 if (locked_balance.amount == 0) {
192 return;
193 }
194
195 // calculate the amount of locked POINTs to convert to liquid POINTs
196 // Warning: these calculations use mixed-type arithmetic. Any changes need to
197 // be thoroughly tested.
198
199 double percentage = unlock_percent / 100.0;
200 double locked_amount = locked_balance.amount / 10000.0;
201 double percentage_applied = locked_amount * percentage;
202 double adjusted_amount = ceil(percentage_applied); // rounding up to whole units
203 uint64_t adjusted_units = adjusted_amount * 10000;
204
205 // to prevent rounding up to more than the locked point balance, apply this adjustment
206 // this will bring the locked balance to zero
207 if (adjusted_units > locked_balance.amount) {
208 adjusted_units = locked_balance.amount;
209 }
210
211 asset converted_points = asset(adjusted_units, POINT_CURRENCY_SYMBOL);
212
213 std::string memo = std::string("unlocking POINTs by ");
214 memo.append(user.to_string());
215
216 // Issue the required amount to the freeos account
217 if (converted_points.amount > 0) {
218 issue(get_self(), converted_points, memo);
219
220 // transfer liquid POINTs to user
221 transfer(get_self(), user, converted_points, memo);
222
223 // subtract the amount transferred from the unvested record
224 locked_accounts_table.modify(locked_account_iterator, get_self(), [&](auto &v) { v.balance -= converted_points; });
225 }
226
227 // record the unlock event in the unvest history table
228 unvest_iterator = unvest_table.begin();
229 if (unvest_iterator == unvest_table.end()) {
230 unvest_table.emplace(get_self(), [&](auto &unvest) {
231 unvest.iteration_number = this_iteration;
232 });
233 } else {
234 unvest_table.modify(unvest_iterator, same_payer, [&](auto &unvest) {
235 unvest.iteration_number = this_iteration;
236 });
237 }
238}

396

413
414 issue_action.send();
415
416 // transfer FREEBI tokens to the owner
417 action transfer_action = action(
418 permission_level{get_self(), "active"_n}, name(freebi_tokens_contract),

420 std::make_tuple(get_self(), owner, exchangeable_amount, memo));

433asset freeosgov::calculate_mint_fee(name &user, asset &mint_quantity, symbol mint_fee_currency) {
434
435 asset mintfee;
436 double mint_fee_percent;
437
438 double mintfee_amount = 0; // amount of currency, e.g. 123.4567 represents 123.4567 FREEOS
439 int64_t mintfee_units = 0; // units of currency, e.g. 1234567 represents 123.4567 FREEOS
440
441 check(mint_quantity.amount > 0, "invalid mint quantity");
442
443 // retrieve the latest mint fee percent from the last reward record
444 rewards_index rewards_table(get_self(), get_self().value);
445 auto reward_iterator = rewards_table.rbegin();
446 check(reward_iterator != rewards_table.rend(), "latest reward record not found");
447
448 // get the latest voted mint fee percent - depending on which currency user is paying with
449 if (mint_fee_currency == symbol(FREEOS_CURRENCY_CODE, 4)) {
450 mint_fee_percent = reward_iterator->mint_fee_percent;
451 } else if (mint_fee_currency == symbol(XPR_CURRENCY_CODE, 4)) {
452 mint_fee_percent = reward_iterator->mint_fee_percent_xpr;
453 } else if (mint_fee_currency == symbol(XUSDC_CURRENCY_CODE, 6)) {
454 mint_fee_percent = reward_iterator->mint_fee_percent_xusdc;
455 }
456
457 // we have to work in double rather asset for accuracy in division and because currencies (e.g. FREEOS, XPR and XUSDC) have different precisions
458 double amount_in_units = mint_quantity.amount / 10000.0;
459 double mintfee_in_freeos = amount_in_units * (mint_fee_percent / 100.0);
460
461 // apply the minimum fee
462 double mintfee_min = get_dparameter(name("mintfeemin"));
463 if (mintfee_in_freeos < mintfee_min) {
464 mintfee_in_freeos = mintfee_min;
465 }
466
467 // apply the currency conversion if necessary
468 if (mint_fee_currency == symbol(FREEOS_CURRENCY_CODE, 4)) {
469 mintfee_units = mintfee_in_freeos * 10000;
470 } else {
471 // conversion required
472
473 // get the FREEOS exchange rate
474 currencies_index currencies_table(get_self(), get_self().value);
475 auto freeos_iterator = currencies_table.find(symbol(FREEOS_CURRENCY_CODE, 4).raw());
476 check(freeos_iterator != currencies_table.end(), "FREEOS currency record not defined");
477 double freeos_rate = freeos_iterator->usdrate;
478
479 if (mint_fee_currency == symbol(XPR_CURRENCY_CODE, 4)) {
480 // get the XPR exchange rate
481 auto xpr_iterator = currencies_table.find(symbol(XPR_CURRENCY_CODE, 4).raw());
482 check(xpr_iterator != currencies_table.end(), "XPR currency record not defined");
483 double xpr_rate = xpr_iterator->usdrate;
484
485 // do the conversion
486 mintfee_amount = round(mintfee_in_freeos * freeos_rate / xpr_rate * 10000);
487 mintfee_units = (int64_t) mintfee_amount;
488 }
489
490 if (mint_fee_currency == symbol(XUSDC_CURRENCY_CODE, 6)) {
491 // get the XUSDC exchange rate
492 auto xusdc_iterator = currencies_table.find(symbol(XUSDC_CURRENCY_CODE, 6).raw());
493 check(xusdc_iterator != currencies_table.end(), "XUSDC currency record not defined");
494 double xusdc_rate = xusdc_iterator->usdrate;
495
496 // do the conversion
497 mintfee_amount = round(mintfee_in_freeos * freeos_rate / xusdc_rate * 1000000);
498 mintfee_units = (int64_t) mintfee_amount;
499 }
500
501 }
502
503 mintfee = asset(mintfee_units, mint_fee_currency); // asset
504
505 // DIAG
506 // check(false, "mintfee_amount = " + to_string(mintfee_amount) + ", mintfee_units = " + to_string(mintfee_units) + ", mint fee = " + mintfee.to_string());
507
508 return mintfee;
509}

529void freeosgov::refund_mintfee(name user, symbol mint_fee_currency) {
530
531 credit_index credit_table(get_self(), user.value);
532 auto credit_iterator = credit_table.find(mint_fee_currency.code().raw());
533
534 if (credit_iterator == credit_table.end()) return; // no credit record, so nothing to refund
535
536 asset mintfee_paid = credit_iterator->balance;
537 symbol mintfee_symbol = mintfee_paid.symbol;
538
539 // look up the currencies table to get the contract name
540 currencies_index currencies_table(get_self(), get_self().value);
541 auto currency_iterator = currencies_table.find(mintfee_symbol.raw());
542
543 if (currency_iterator == currencies_table.end()) return; // unknown currency
544 name currency_contract = currency_iterator->contract;
545
546 // transfer the fee back to the user
547 action transfer_action = action(
548 permission_level{get_self(), "active"_n}, currency_contract,
549 "transfer"_n,
550 std::make_tuple(get_self(), user, mintfee_paid, string("refund of incorrect freeos mint fee")));
551
552 transfer_action.send();
553
554 // delete the credit record
555 credit_table.erase(credit_iterator);
556}

569bool freeosgov::process_mint_fee(name user, asset mint_quantity, symbol mint_fee_currency) {
570
571 bool mintfee_status; // will be set to true if correct mint fee has been paid
572
573 // calculate the mint fee
574 asset mintfee = calculate_mint_fee(user, mint_quantity, mint_fee_currency);
575
576 // has the user paid the mint fee, i.e. got credit?
577 asset user_credit = asset(0, mint_fee_currency); // default
578 credit_index credit_table(get_self(), user.value);
579 auto credit_iterator = credit_table.find(mint_fee_currency.code().raw());
580
581 if (credit_iterator != credit_table.end()) {
582 user_credit = credit_iterator->balance;
583 }
584
585 // Check if the mint-fee paid is the right amount
586 if (mintfee == user_credit) {
587 // correct amount
588 // erase the credit record (if paid)
589 if (credit_iterator != credit_table.end()) {
590 credit_table.erase(credit_iterator);
591 }
592
593 mintfee_status = true;
594 } else {
595 // incorrect amount - refund the incorrect mint fee
596 refund_mintfee(user, mint_fee_currency);
597 mintfee_status = false;
598 }
599
600 return mintfee_status;
601}

670void freeosgov::mintfreeos(name user, const asset &input_quantity, symbol &mint_fee_currency, bool use_airclaim_points) {
671
672 require_auth(user);
673
674 // check that system is operational (masterswitch parameter set to "1")
675 check(check_master_switch(), MSG_FREEOS_SYSTEM_NOT_AVAILABLE);
676
677 uint32_t this_iteration = current_iteration();
678
679 // is the system operational?
680 check(this_iteration != 0, "The freeos system is not available at this time");
681
682 check(input_quantity.is_valid(), "invalid quantity");
683 check(input_quantity.amount > 0, "must mint a positive quantity");
684
685 // check that the user has the appropriate POINTs balance to mint with
686 // We don't need to do this for minting from FREEBI because the FREEBI is sent as a transfer credit
687 if (input_quantity.symbol == POINT_CURRENCY_SYMBOL) {
688 // get POINT balance
689 auto points_balance_amount = 0; // default value
690 accounts points_accounts_table(get_self(), user.value);
691 auto points_iterator = points_accounts_table.find(POINT_CURRENCY_SYMBOL.code().raw());
692 if (points_iterator != points_accounts_table.end()) {
693 points_balance_amount = points_iterator->balance.amount;
694 }
695
696 check(points_balance_amount >= input_quantity.amount, "user has insufficient POINTs balance");
697 }
698
699 if (use_airclaim_points == false) {
700
701 if (has_nft(user) == false) { // users with an active NFT do not pay the mint fee
702 // check whether user has paid correct mint fee, whether they have a credit record, adjust their mintfeefree allowance
703 check(process_mint_fee(user, input_quantity, mint_fee_currency) == true, "incorrect mint fee has been paid");
704 }
705
706 } else {
707 // check mff balance, then decrease by the appropriate amount of POINTs minted
708
709 // get the user's mff balance
710 asset mff_balance = asset(0, POINT_CURRENCY_SYMBOL); // default value
711 mintfeefree_index mff_table(get_self(), user.value);
712 auto mff_iterator = mff_table.begin();
713 if (mff_iterator != mff_table.end()) {
714 mff_balance = mff_iterator->balance;
715 }
716
717 // check whether the user has enough mff POINTs
718 check(input_quantity.amount <= mff_balance.amount, "insufficient Airclaim points to mint the requested amount");
719
720 // decrease the mff balance
721 mff_table.modify(mff_iterator, get_self(),
722 [&](auto &mff) { mff.balance -= input_quantity; });
723 }
724
725
726 // different processing required for input_quantity currencies
727 if (input_quantity.symbol == POINT_CURRENCY_SYMBOL) {
728 adjust_balances_from_points(user, input_quantity);
729 } else if (input_quantity.symbol == FREEBI_CURRENCY_SYMBOL) {
730 adjust_balances_from_freebi(user, input_quantity);
731 } else {
732 check(false, "invalid currency for input quantity");
733 }
734
735 // Issue FREEOS
736 asset exchangeable_amount = asset(input_quantity.amount, FREEOS_CURRENCY_SYMBOL);
737 std::string memo = std::string("minting");
738
739 // ask FREEOS contract to issue an equivalent amount of FREEOS tokens to the freeosgov account
740 string freeos_tokens_contract = get_parameter(name("freeostokens"));
741 action issue_action = action(
742 permission_level{get_self(), "active"_n}, name(freeos_tokens_contract),
743 "issue"_n, std::make_tuple(get_self(), exchangeable_amount, memo));
744
745 issue_action.send();
746
747 // transfer FREEOS tokens to the user
748 action transfer_action = action(
749 permission_level{get_self(), "active"_n}, name(freeos_tokens_contract),
750 "transfer"_n,
751 std::make_tuple(get_self(), user, exchangeable_amount, memo));
752
753 transfer_action.send();
754}

762void freeosgov::withdraw(const name user) {
763 action transfer_action;
764 asset credit_amount;
765 name currency_contract;
766 string memo;
767
768 require_auth(user);
769
770 // check that system is operational (global masterswitch parameter set to "1")
771 check(check_master_switch(), MSG_FREEOS_SYSTEM_NOT_AVAILABLE);
772
773 // currencies table
774 currencies_index currencies_table(get_self(), get_self().value);
775
776 // credits table
777 credit_index credits_table(get_self(), user.value);
778 auto credit_iterator = credits_table.begin();
779
780 while (credit_iterator != credits_table.end()) {
781 // amount
782 credit_amount = credit_iterator->balance;
783
784 // get the currency contract
