96string get_account_type(name user) {
97 // default result
98 string user_account_type = "e";
99
100 // first determine which contract we consult - if we have set an alternative
101 // contract then use that one
102 name verification_contract = VERIFICATION_CONTRACT;
103
104 // access the verification table
105 usersinfo verification_table(name(verification_contract),
106 name(verification_contract).value);
107 auto verification_iterator = verification_table.find(user.value);
108
109 if (verification_iterator != verification_table.end()) {
110 // record found, so default account_type is 'd', unless we find a
111 // verification
112 user_account_type = "d";
113
114 auto kyc_prov = verification_iterator->kyc;
115
116 for (int i = 0; i < kyc_prov.size(); i++) {
117 size_t fn_pos = kyc_prov[i].kyc_level.find("firstname");
118 size_t ln_pos = kyc_prov[i].kyc_level.find("lastname");
119
120 if (fn_pos != std::string::npos && ln_pos != std::string::npos) {
121 user_account_type = "v";
122 break;
123 }
124 }
125 }
126
127 return user_account_type;
128}

136asset freeosgov::calculate_user_cls_addition() {
137 // get parameters
138 parameters_index parameters_table(get_self(), get_self().value);
139
140 // user cls contribution
141 int64_t uclsamount;
142 auto uclsamount_iterator = parameters_table.find(name("uclsamount").value);
143 if (uclsamount_iterator != parameters_table.end()) {
144 uclsamount = stoi(uclsamount_iterator->value);
145 } else {
146 uclsamount = UCLSAMOUNT; // hard floor constant
147 }
148
149 // get double parameters
150 dparameters_index dparameters_table(get_self(), get_self().value);
151
152 // daoshare
153 double daoshare;
154 auto daoshare_iterator = dparameters_table.find(name("daoshare").value);
155 if (daoshare_iterator != dparameters_table.end()) {
156 daoshare = daoshare_iterator->value;
157 } else {
158 daoshare = DAOSHARE; // hard floor constant
159 }
160
161 // partnershare
162 double partnershare;
163 auto partnershare_iterator = dparameters_table.find(name("partnershare").value);
164 if (partnershare_iterator != dparameters_table.end()) {
165 partnershare = partnershare_iterator->value;
166 } else {
167 partnershare = PARTNERSHARE; // hard floor constant
168 }
169
170 // calculate the total asset for user CLS
171 int64_t user_cls_amount = uclsamount * (1.0 + daoshare + partnershare);
172 asset ucls = asset(user_cls_amount * 10000, POINT_CURRENCY_SYMBOL);
173
174 return ucls;
175}

188void freeosgov::reguser(name user) { // TODO: detect if the user has an existing record from the airclaim

193 check(check_master_switch(), MSG_FREEOS_SYSTEM_NOT_AVAILABLE);
194
195 // get the current iteration
196 uint32_t iteration = current_iteration();
197
198 check(iteration != 0, "The freeos system is not yet available");
199
200 // is the user already registered?
201 // find the account in the participants table
202 participants_index participants_table(get_self(), user.value);
203 auto participant_iterator = participants_table.begin();
204
205 check(participant_iterator == participants_table.end(), "user is already registered");
206
207 // determine account type
208 string account_type = get_account_type(user);
209 check(account_type == "v" || account_type == "b" || account_type == "c", "please complete kyc before registering");
210
211 // capture the user's POINTs balance - as these POINTs will be mint-fee-free
212 asset liquid_points = asset(0, POINT_CURRENCY_SYMBOL); // default=0 if POINTs balance record not found
213 accounts accounts_table(get_self(), user.value);
214 auto points_iterator = accounts_table.find(symbol_code(POINT_CURRENCY_CODE).raw());
215 if (points_iterator != accounts_table.end()) {
216 liquid_points = points_iterator->balance;
217 }
218
219 // also include the locked POINTs balance
220 asset locked_points = asset(0, POINT_CURRENCY_SYMBOL); // default=0 if locked POINTs balance record not found
221 lockaccounts locked_accounts_table(get_self(), user.value);
222 auto locked_points_iterator = locked_accounts_table.find(symbol_code(POINT_CURRENCY_CODE).raw());
223 if (locked_points_iterator != locked_accounts_table.end()) {
224 locked_points = locked_points_iterator->balance;
225 }
226
227 // determine if the user has an AIRKEY
228 asset airkey_allowance = asset(0, POINT_CURRENCY_SYMBOL); // default=0 if no AIRKEY
229 auto airkey_iterator = accounts_table.find(symbol_code(AIRKEY_CURRENCY_CODE).raw());
230 if (airkey_iterator != accounts_table.end()) {
231 airkey_allowance = asset(AIRKEY_MINT_FEE_FREE_ALLOWANCE * 10000, POINT_CURRENCY_SYMBOL);
232 }
233
234 asset mintfeefree_allowance = liquid_points + locked_points + airkey_allowance;
235
236 // store the mint-fee-free allowance
237 if (mintfeefree_allowance.amount > 0) {
238
239 // store in the mint_fee_free table
240 mintfeefree_index mintfeefree_table(get_self(), user.value);
241 auto mintfeefree_iterator = mintfeefree_table.begin();
242
243 if (mintfeefree_iterator == mintfeefree_table.end()) {
244 // emplace
245 mintfeefree_table.emplace(get_self(), [&](auto &m) {
246 m.balance = mintfeefree_allowance;
247 });
248 } else {
249 // modify - should not be necessary to modify, but include this code in the event of migrations, etc
250 mintfeefree_table.modify(mintfeefree_iterator, get_self(), [&](auto &m) {
251 m.balance = mintfeefree_allowance;
252 });
253 }
254 }
255
256 // add record to the participants table
257 participants_table.emplace(get_self(), [&](auto &participant) {
258 participant.account_type = account_type;
259 participant.registered_iteration = iteration;
260 participant.total_issuance_amount = asset(0, POINT_CURRENCY_SYMBOL);
261 });
262
263 // update the system record - number of users and CLS
264 asset ucls = calculate_user_cls_addition();
265
266 system_index system_table(get_self(), get_self().value);
267 auto system_iterator = system_table.begin();
268 if (system_iterator == system_table.end()) {
269 // emplace
270 system_table.emplace(
271 get_self(), [&](auto &sys) {
272 sys.usercount = 1;
273
274 // update the CLS if a verified user
275 if (is_user_verified(user)) {
276 sys.cls = ucls;
277 }
278
279 });
280 } else {
281 // modify
282 system_table.modify(system_iterator, get_self(), [&](auto &sys) {
283 sys.usercount += 1;
284
285 // update the CLS if a verified user
286 if (is_user_verified(user)) {
287 sys.cls += ucls;
288 }
289 });
290 }

303void freeosgov::reregister(name user) {
304 require_auth(user);
305
306 // check that system is operational (masterswitch parameter set to "1")
307 check(check_master_switch(), MSG_FREEOS_SYSTEM_NOT_AVAILABLE);
308
309 // original verified user status
310 bool verified_before = is_user_verified(user);
311
312 // get the current iteration
313 check(current_iteration() != 0, "The freeos system is not yet available");
314
315 // set the account type
316 participants_index participants_table(get_self(), user.value);
317 auto participant_iterator = participants_table.begin();
318
319 // check if the user has a user registration record

321
322 // get the account type
323 string account_type = get_account_type(user);
324
325 // set the participant account type
326 participants_table.modify(participant_iterator, get_self(), [&](auto &participant) {
327 participant.account_type = account_type;
328 });
329
330 // new verified user status
331 bool verified_after = is_user_verified(user);
332
333 // update the CLS if a user has become verified

336 auto system_iterator = system_table.begin();
337 check(system_iterator != system_table.end(), "system record is undefined");
338
339 // modify cls

341
342 system_table.modify(system_iterator, get_self(), [&](auto &sys) {
