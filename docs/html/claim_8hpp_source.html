24void freeosgov::claim(name user) {
25
26 require_auth(user);
27
28 // check that system is operational (masterswitch parameter set to "1")
29 check(check_master_switch(), MSG_FREEOS_SYSTEM_NOT_AVAILABLE);
30
31 tick();
32
33 uint32_t this_iteration = current_iteration();
34
35 // is the system operational?
36 check(this_iteration != 0, "The freeos system is not available at this time");
37
38 // find the participant's last claim
39 participants_index participants_table(get_self(), user.value);
40 auto participant_iterator = participants_table.begin();
41 check(participant_iterator != participants_table.end(), "user registration record is undefined");
42 uint32_t last_claim = participant_iterator->last_claim;
43
44 // find the user's svr record
45 svr_index svr_table(get_self(), user.value);
46 auto svr_iterator = svr_table.begin();
47 check(svr_iterator != svr_table.end(), "user has not completed any votes or surveys");
48
49 // check if user has already claimed in this iteration
50 check(this_iteration != last_claim, "you cannot claim more than once in an iteration");
51
52 // check that we are not in iteration 1 - payments can only be made retrospectively
53 check(this_iteration > 1, "claims are for previous iterations only");
54
55 // get the current proportion of locked POINTs
56 double locked_proportion = get_locked_proportion();
57
58 // determine which range of iterations we are paying for
59 uint32_t earliest_payment_iteration = this_iteration > 4 ? this_iteration - 4 : 1;
60 uint32_t latest_payment_iteration = this_iteration - 1;
61
62 // open the rewards table
63 rewards_index rewards_table(get_self(), get_self().value);
64
65 // get the S,V,R share values
66 double survey_share = get_dparameter(name("surveyshare"));
67 double vote_share = get_dparameter(name("voteshare"));
68 double ratify_share = get_dparameter(name("ratifyshare"));
69
70 double freedaoshare = get_dparameter(name("freedaoshare"));
71 double partnershare = get_dparameter(name("partnershare"));
72
73 // get the freeosdiv and partners accounts
74 name freeosdiv_acct = name(get_parameter(name("freedaoacct")));
75 name partners_acct = name(get_parameter(name("partnersacct")));
76
77 // keep some counters
78 asset total_user_payment = asset(0, POINT_CURRENCY_SYMBOL);
79 asset grand_total_iterations_payments = asset(0, POINT_CURRENCY_SYMBOL);
80 uint8_t total_issuances = 0;
81
82
83 // determine each iteration's payments
84 for (uint32_t iter = earliest_payment_iteration; iter <= latest_payment_iteration; iter++) {
85
86 // has the user already claimed for this iteration?
87 // developer note: this next statement could be incorporated into the for loop, but implementing this way for clarity
88 if (iter <= last_claim) continue;
89
90 // get the reward record for the iteration
91 auto reward_iterator = rewards_table.find(iter);
92 if (reward_iterator == rewards_table.end()) continue; // no reward record so continue to next iteration
93
94 // check if the vote was ratified, otherwise no payment to be made for that iteration
95 if (reward_iterator->ratified == false) continue;
96
97 // running total of user payments amounts
98 uint64_t user_payment_amount = 0;
99
100 // initialise the memo
101 string transfer_memo = "Claim for week " + to_string(iter) + ": ";
102
103 // determine the total user payout for the iteration
104 uint64_t iteration_reward_amount = reward_iterator->participant_issuance.amount;
105
106 // split the total user payout according to whether S, V or R completed
107 uint64_t payment_survey_amount = iteration_reward_amount * survey_share;
108 uint64_t payment_vote_amount = iteration_reward_amount * vote_share;
109 uint64_t payment_ratify_amount = iteration_reward_amount * ratify_share;
110
111 // check if survey completed for this iteration
112 transfer_memo += "Survey [" + to_string((int)(survey_share * 100)) + "%]: ";
113 if (svr_iterator->survey0 == iter ||
114 svr_iterator->survey1 == iter ||
115 svr_iterator->survey2 == iter ||
116 svr_iterator->survey3 == iter ||
117 svr_iterator->survey4 == iter) {
118 user_payment_amount += payment_survey_amount;
119 transfer_memo += "(" + asset(payment_survey_amount, POINT_CURRENCY_SYMBOL).to_string() + ")";
120 } else {
121 transfer_memo += "(not completed)";
122 }
123
124 // check if vote completed for this iteration
125 transfer_memo += ", Vote [" + to_string((int)(vote_share * 100)) + "%]: ";
126 if (svr_iterator->vote0 == iter ||
127 svr_iterator->vote1 == iter ||
128 svr_iterator->vote2 == iter ||
129 svr_iterator->vote3 == iter ||
130 svr_iterator->vote4 == iter) {
131 user_payment_amount += payment_vote_amount;
132 transfer_memo += "(" + asset(payment_vote_amount, POINT_CURRENCY_SYMBOL).to_string() + ")";
133 } else {
134 transfer_memo += "(not completed)";
135 }
136
137 // check if ratify completed for this iteration
138 transfer_memo += ", Ratify [" + to_string((int)(ratify_share * 100)) + "%]: ";
139 if (svr_iterator->ratify0 == iter ||
140 svr_iterator->ratify1 == iter ||
141 svr_iterator->ratify2 == iter ||
142 svr_iterator->ratify3 == iter ||
143 svr_iterator->ratify4 == iter) {
144 user_payment_amount += payment_ratify_amount;
145 transfer_memo += "(" + asset(payment_ratify_amount, POINT_CURRENCY_SYMBOL).to_string() + ")";
146 } else {
147 transfer_memo += "(not completed)";
148 }
149
150 // make the payment, if any
151 if (user_payment_amount > 0) {
152 total_issuances++;
153
154 uint64_t user_locked_amount = user_payment_amount * locked_proportion;
155 uint64_t user_liquid_amount = user_payment_amount - user_locked_amount;
156
157 // get the partner and freedao shares
158 uint64_t freedao_payment_amount = user_payment_amount * freedaoshare;
159 uint64_t partners_payment_amount = user_payment_amount * partnershare;
160
161 // calculate the payments as assets
162 asset user_payment = asset(user_payment_amount, POINT_CURRENCY_SYMBOL);
163 asset user_liquid_payment = asset(user_liquid_amount, POINT_CURRENCY_SYMBOL);
164 asset user_locked_payment = asset(user_locked_amount, POINT_CURRENCY_SYMBOL);
165 asset freedao_payment = asset(freedao_payment_amount, POINT_CURRENCY_SYMBOL);
166 asset partners_payment = asset(partners_payment_amount, POINT_CURRENCY_SYMBOL);
167
168 total_user_payment += user_payment;
169
170 // mint the total amount
171 asset iteration_all_payments = user_payment + freedao_payment + partners_payment;
172 grand_total_iterations_payments += iteration_all_payments;
173
174 string issue_memo = string("claim by ") + user.to_string() + " for week " + to_string(iter);
175
176 // issue all POINTs
177 action issue_action = action(
178 permission_level{get_self(), "active"_n}, get_self(),
179 "mint"_n, std::make_tuple(get_self(), get_self(), iteration_all_payments, issue_memo));
180 issue_action.send();
181
182 // explain the unlocked/locked amounts in the memo
183 transfer_memo += " - payment (unlocked/locked) = " + user_liquid_payment.to_string() + "/" + user_locked_payment.to_string();
184
185 // transfer liquid POINTs to the user account
186 action user_transfer_action = action(
187 permission_level{get_self(), "active"_n}, get_self(),
188 "allocate"_n,
189 std::make_tuple(get_self(), user, user_liquid_payment, transfer_memo));
190 user_transfer_action.send();
191
192 // increase the user's locked balance
193 if (user_locked_amount > 0) {
194 lockaccounts lock_accounts(get_self(), user.value);
195 auto acct_iterator = lock_accounts.find(user_locked_payment.symbol.code().raw());
196 if (acct_iterator == lock_accounts.end()) {
197 lock_accounts.emplace(get_self(), [&](auto &a) { a.balance = user_locked_payment; });
198 } else {
199 lock_accounts.modify(acct_iterator, get_self(), [&](auto &a) { a.balance += user_locked_payment; });
200 }
201 }
202
203 // memo for freedao and partners shares
204 string shares_memo = string("share of claim by ") + user.to_string() + " for week " + to_string(iter);
205
206 // transfer POINTs to the freedao account
207 // first, record the deposit to the freedao account
208 record_deposit(this_iteration, freedao_payment);
209
210 if (freedao_payment_amount > 0) {
211 action freedao_transfer_action = action(
212 permission_level{get_self(), "active"_n}, get_self(),
213 "allocate"_n,
214 std::make_tuple(get_self(), freeosdiv_acct, freedao_payment, shares_memo));
215 freedao_transfer_action.send();
216 }
217
218 // transfer POINTs to the partners account
219 if (partners_payment_amount > 0) {
220 action partners_transfer_action = action(
221 permission_level{get_self(), "active"_n}, get_self(),
222 "allocate"_n,
223 std::make_tuple(get_self(), partners_acct, partners_payment, shares_memo));
224 partners_transfer_action.send();
225 }
226 }
227 }
228
229 // update the participant record issuance values
230 participants_table.modify(participant_iterator, get_self(), [&](auto &participant) {
231 participant.last_claim = latest_payment_iteration; // i.e. paid up to this iteration
232 participant.total_issuance_amount += total_user_payment;
233 participant.issuances += total_issuances;
234 });
235
236
237 // update the number of claimevents in the system record and reduce CLS
238 if (total_issuances > 0) {
239 system_index system_table(get_self(), get_self().value);
240 auto system_iterator = system_table.begin();
241 check(system_iterator != system_table.end(), "system record is undefined");
242 system_table.modify(system_iterator, get_self(), [&](auto &s) {
243 s.claimevents += total_issuances;
244 s.cls -= grand_total_iterations_payments; // reduce CLS by total number of points issued
245 });
246 }
