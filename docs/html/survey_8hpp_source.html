25vector<string> split (string s, string delimiter) {
26 size_t pos_start = 0, pos_end, delim_len = delimiter.length();
27 string token;
28 vector<string> res;
29
30 while ((pos_end = s.find (delimiter, pos_start)) != string::npos) {
31 token = s.substr (pos_start, pos_end - pos_start);
32 pos_start = pos_end + delim_len;
33 res.push_back (token);
34 }
35
36 res.push_back (s.substr (pos_start));
37 return res;
38}

49std::vector<int> parse_survey_ranges(string surveyranges) {
50
51 // the surveyranges string looks like this: q2:1-48,q4:1-48
52 std::vector<int> limits;
53
54 std::vector<std::string> tokenlist = split(surveyranges, ",");
55
56 std::vector q1_2_param = split(tokenlist[0], ":");
57 std::vector q2_2_param = split(tokenlist[1], ":");
58
59 std::vector q1_minmax = split(q1_2_param[1], "-");
60 std::vector q2_minmax = split(q2_2_param[1], "-");
61
62 limits.push_back (stoi(q1_minmax[0]));
63 limits.push_back (stoi(q1_minmax[1]));
64 limits.push_back (stoi(q2_minmax[0]));
65 limits.push_back (stoi(q2_minmax[1]));
66
67 return limits;
68}
